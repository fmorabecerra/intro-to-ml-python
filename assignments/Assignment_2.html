def answer_two():
    from sklearn.linear_model import LinearRegression
    from sklearn.preprocessing import PolynomialFeatures
    from sklearn.metrics.regression import r2_score

    # Your code here
    poly_order =  range(10)

    # Create your arrays where you will eventually stash results.
    r2_train = np.zeros(10)
    r2_test = np.zeros(10)
    
    # do everything in a loop so that your code is cleaner
    for i in range(len(poly_order)):
        
        # Create your object that you will use to expland your feature space.
        poly = PolynomialFeatures(degree = poly_order[i])
        # Expand your feature space by expanding your input
        X_train_poly = poly.fit_transform(X_train.reshape(len(X_train), 1))
    
        # Train you linear regression model with the expanded nfeature space.
        linreg = LinearRegression().fit(X_train_poly, y_train)

        # You are going to need to transform the test data as well. 
        # Any transformation on input must be done on output.
        X_test_poly = poly.fit_transform(X_test.reshape(len(X_test), 1))

        # Compute the R-squared score
        r2_train[i,] = linreg.score(X_train_poly, y_train)
        r2_test[i,] = linreg.score(X_test_poly, y_test)        

    return (r2_train, r2_test) # Convert data to tuple
